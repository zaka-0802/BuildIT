            // Add message
            const messageSpan = document.createElement('span');
            messageSpan.className = 'text-lg font-medium';
            messageSpan.textContent = message;

            // Add to document

        // Load a specific program
        async function loadProgram(id) {
            try {
                const response = await fetch(`/api/program/${id}`);
                const program = await response.json();
                
                document.getElementById('programName').value = program.name;
                document.getElementById('codeOutput').textContent = program.pythonCode;
                
                // Load the XML into the Blockly workspace
                const xml = Blockly.utils.xml.textToDom(program.xml);
                Blockly.Xml.clearWorkspaceAndLoadFromXml(xml, workspace);
                
                showNotification(`Loaded program: ${program.name}`, 'success');
            } catch (error) {
                console.error('Error loading program:', error);
                showNotification('Could not load your program', 'error');
            }
        }

        // Convert blocks to Python code
        document.getElementById('convertBtn').addEventListener('click', async () => {
            const xml = Blockly.Xml.workspaceToDom(workspace);
            const xmlText = Blockly.Xml.domToText(xml);

            try {
                showNotification('Converting your blocks to code...', 'info');
                
                const response = await fetch('/api/convert', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ xml: xmlText }),
                });

                const pythonCode = await response.text();
                console.log("Python Code Before Saving:", pythonCode);

                document.getElementById('codeOutput').textContent = pythonCode;
                showNotification('Your blocks have been converted to code!', 'success');
            } catch (error) {
                console.error('Error converting code:', error);
                showNotification('Could not convert your blocks to code', 'error');
            }
        });

        // Save program
        document.getElementById('saveBtn').addEventListener('click', async () => {
            const name = document.getElementById('programName').value;
            if (!name) {
                showNotification('Please give your robot program a name!', 'warning');
                document.getElementById('programName').focus();
                return;
            }

            const xml = Blockly.Xml.workspaceToDom(workspace);
            const xmlText = Blockly.Xml.domToText(xml);
            const pythonCode = document.getElementById('codeOutput').textContent;

            try {
                showNotification('Saving your robot program...', 'info');
                
                const response = await fetch('/api/save', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        name: name,
                        xml: xmlText,
                        pythonCode: pythonCode
                    }),
                });

                const result = await response.json();
                
                if (result.success) {
                    loadPrograms();  // Refresh the program list
                    showNotification('Your robot program has been saved!', 'success');
                }
            } catch (error) {
                console.error('Error saving program:', error);
                showNotification('Could not save your robot program', 'error');
            }
        });

        //Execute program
        document.getElementById('executeBtn').addEventListener('click', async () => {
    const code = document.getElementById('codeOutput').textContent.trim();
    if (!code) {
        showNotification('First convert your blocks to code!', 'warning');
        return;
    }

    try {
        showNotification('Running your robot program...', 'info');

        const response = await fetch('/api/execute', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({ code }),
        });

        const result = await response.json();

        if (result.success) {
            // Success case
            showNotification(result.message || 'Your robot program is running!', 'success');
        } else {
            // Error case - check if it's a connection error
            const errorMessage = result.message || 'Failed to run your robot program';
            
            // If it's a connection error (contains "Failed to send" or similar phrases)
            if (errorMessage.includes('Failed to send') || 
                errorMessage.includes('Connection') || 
                errorMessage.includes('connect') ||
                errorMessage.includes('ssh') ||
                errorMessage.includes('scp')) {
                
                showErrorModal(errorMessage);
            } else {
                // For other errors, use the regular notification
                showNotification(errorMessage, 'error');
            }
        }
    } catch (error) {
        console.error('Execution error:', error);
        showErrorModal('Could not connect to the server. Please try again.');
    }
});

        function showNotification(message, type = 'info') {
            // Create notification element

            notification.className = 'fixed bottom-4 right-4 px-6 py-4 rounded-xl shadow-lg flex items-center space-x-3 transition-all transform translate-y-0 opacity-100';
            
            // Set background color based on type
            switch(type) {
                case 'success':
                    notification.classList.add('bg-success-500', 'text-white');
                    notification.innerHTML = `
                        <svg xmlns="http://www.w3.org/2000/svg" class="h-6 w-6" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">

                    break;
                case 'warning':
                    notification.classList.add('bg-warning-500', 'text-white');
                    notification.innerHTML = `
                        <svg xmlns="http://www.w3.org/2000/svg" class="h-6 w-6" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">

                    break;
                case 'error':
                    notification.classList.add('bg-danger-500', 'text-white');
                    notification.innerHTML = `
                        <svg xmlns="http://www.w3.org/2000/svg" class="h-6 w-6" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">

                        </svg>
                    `;
                    break;
                default: // info
                    notification.classList.add('bg-primary-500', 'text-white');
                    notification.innerHTML = `
                        <svg xmlns="http://www.w3.org/2000/svg" class="h-6 w-6" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">

                            <line x1="12" y1="8" x2="12.01" y2="8"></line>
                        </svg>
                    `;
            }
            
            // Add message
            const messageSpan = document.createElement('span');
            messageSpan.className = 'text-lg font-medium';
            messageSpan.textContent = message;
            notification.appendChild(messageSpan);
            
            // Add to document
            document.body.appendChild(notification);
            
            // Remove after 3 seconds
            setTimeout(() => {
                notification.classList.add('opacity-0', 'translate-y-2');
                setTimeout(() => {
                    notification.remove();
                }, 300);
            }, 3000);
        }

@app.route('/api/save', methods=['POST'])
def save_program():
    """Save a program, updating it if the name already exists."""
    try:
        data = request.json
        if not data or not all(k in data for k in ['name', 'xml', 'pythonCode']):
            return jsonify({'error': 'Missing required fields'}), 400

        # Check for existing program with the same name
        existing_program = None
        for filename in os.listdir(PROGRAMS_DIR):
            if filename.endswith('.json'):
                file_path = os.path.join(PROGRAMS_DIR, filename)
                with open(file_path, 'r') as f:
                    program_data = json.load(f)
                    if program_data.get("name") == data["name"]:
                        existing_program = file_path
                        break  # Stop searching once found

        # If program exists, overwrite the file
        if existing_program:
            file_path = existing_program
        else:
            # Otherwise, create a new file with a unique ID
            program_id = str(int(datetime.now().timestamp()))
            file_path = os.path.join(PROGRAMS_DIR, f"{program_id}.json")

        # Save the updated or new program data
        program_data = {
            'id': os.path.splitext(os.path.basename(file_path))[0],  # Extract file ID
            'name': data['name'],
            'xml': data['xml'],
            'pythonCode': data['pythonCode'],
            'lastModified': datetime.now().isoformat()
        }

        with open(file_path, 'w') as f:
            json.dump(program_data, f, indent=2)

        return jsonify({'success': True, 'id': program_data['id']})

    except Exception as e:
        return jsonify({'error': str(e)}), 500

import xml.etree.ElementTree as ET

def generate_python_code_from_blockly_xml(xml_string):
    """
    Generate Python code from the XML representation of Blockly.
    
    Args:
        xml_string: XML string exported by Blockly.
    Returns:
        Generated Python code as a string.
    """
    try:
        # Parse the XML
        root = ET.fromstring(xml_string)
        python_code = []  # Store code lines in a list for better management
        print("XML Root:", ET.tostring(root, encoding="unicode"))

        def parse_block(block):
            if block is None:
             return "100"

            namespace = {'ns0': 'https://developers.google.com/blockly/xml'}  # Namespace definition
            block_type = block.attrib.get("type")
            print(f"Processing Block: {block_type}")  # Debugging output

            # FIX: Extract fields, values, and statements correctly
            fields = {field.attrib["name"]: field.text for field in block.findall("ns0:field", namespace)}
            values = {}
            
            for value in block.findall("ns0:value", namespace):
                name = value.attrib["name"]
                block_or_shadow = value.find("ns0:block", namespace) or value.find("ns0:shadow", namespace)
                if block_or_shadow is not None:
                    values[name] = parse_block(block_or_shadow)
            statements = {}
            for statement in block.findall("ns0:statement", namespace):
                name = statement.attrib["name"]
                statements[name] = parse_block(statement.find("ns0:block", namespace))  # Recursively parse blocks inside statements

            next_block = block.find("ns0:next/ns0:block", namespace)

            print(f"Fields: {fields}")  # Debugging
            print(f"Values: {values}")  # Debugging
            print(f"Statements: {statements}")  # Debugging
            print(f"Next: {next_block}")  # Debugging
            code_lines = []
